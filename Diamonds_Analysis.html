Carat and Price.

data(diamonds)
ggplot(data = diamonds, aes(x = carat, y = price)) +
  geom_point() +
  coord_cartesian(xlim = c(0, quantile(diamonds$carat, .99)),
                  ylim = c(0, quantile(diamonds$price, .99))) +
  geom_smooth(method = 'lm')
